<script>
function light(value)
	{
	var pic;
		if(value == 0) 
		{
			pic="light off.png";
		}
		else
		{
			pic="light on.png";
		}
		document.getElementById('bulb').src=pic;
	}
</script>


<img id="bulb">

<br>
<button onclick="light(1)">Turn ON</button>
<button onclick="light(0)">Turn OFF</button>
